```md
<script setup>
import * as checkbox from "@zag-js/checkbox"
import { useMachine, normalizeProps } from "@zag-js/vue"
import { computed } from "vue"

const [state, send] = useMachine(checkbox.machine)
const ref = useSetup({ send, id: "1" })
const api = computed(() => checkbox.connect(state.value, send, normalizeProps))
</script>

<template>
  <label ref="ref" v-bind="api.rootProps">
    <span v-bind="api.labelProps">
      Input is
      <span v-if="api.isChecked"> checked</span>
      <span v-else> unchecked</span>
    </span>
    <input v-bind="api.inputProps" />
    <div v-bind="api.controlProps" />
  </label>
</template>
```
